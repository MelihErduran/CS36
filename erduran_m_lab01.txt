/* This program gets two user integers and does multiple operations */
/*                                                                  */
/********************************************************************/

#include<stdio.h>

int main()
{
    //Declaration
    int num1, num2, sum, product, difference, remainder, quotient;
    
    //Data/input
    printf("Enter two numbers: ");
    scanf("%d%d", &num1, &num2);

    //Processing/Calculations
    sum = num1 + num2;
    difference = num1 - num2;
    product = num1 * num2;
    quotient = num1 / (float)num2;
    remainder = num1 % num2;

    //Output
    printf("The sum is %d\n", sum);
    printf("The product is %d\n", product);
    printf("The difference is %d\n", difference);
    printf("The quotient is %d\n", quotient);
    printf("The remainder is %d\n", remainder);

    return 0;
}

/* This program gets the user input of the radius of the circle and print out information about the circle */
/*                                                                                                         */
/***********************************************************************************************************/

#include <stdio.h>

int main() {
    // Declaration
    const float PI = 3.14159;
    int radius;
    float diameter, area, circumference;

    // Data/Input
    printf("The radius of the circle is ");
    scanf("%d", &radius);

    // Processing/Calculations
    diameter = radius * 2;
    area = PI * (radius * radius);
    circumference = (PI * radius) * 2;

    // Output
    printf("The diameter of the circle is %f\n", diameter);
    printf("The area of the circle is %f\n", area);
    printf("The circumference of the circle is %f\n", circumference);

    return 0;
}

/* This program gets the sum of two user-inputted fractions */
/*                                                          */
/************************************************************/

#include <stdio.h>

int main()
{
    //Declaration
    int numerator1, denominator1, numerator2, denominator2, result_numerator, result_denominator;

    //Data/Input
    printf("Enter First Fraction : ");
    scanf("%d/%d", &numerator1, &denominator1);
    printf("Enter the second fraction : ");
    scanf("%d/%d", &numerator2, &denominator2);

    //Processing/Calculations
    result_numerator = (numerator1 * denominator2) + (numerator2 * denominator1);
    result_denominator = denominator1 * denominator2;

    //Output
    printf("\n");
    printf("The sum of the two fractions is %d/%d\n", result_numerator, result_denominator);

    return 0;
}
